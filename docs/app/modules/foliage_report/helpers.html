  35class LeyLiebig:
  36    """
  37    Clase que implementa la Ley del Mínimo de Liebig para el cálculo de nutrientes en un cultivo.
  38    
  39    La Ley del Mínimo establece que el crecimiento de una planta está limitado por el nutriente más escaso en relación
  40    con sus necesidades, en lugar de depender de la cantidad total de nutrientes disponibles.
  41    
  42    Esta clase permite evaluar el estado nutricional de un cultivo comparando los niveles actuales con la demanda ideal,
  43    identificando el nutriente más limitante y proponiendo ajustes para optimizar su disponibilidad.
  44    """
  45
  46    def __init__(self, nutrientes: dict, demanda_planta: Decimal):
  47        """
  48        Inicializa la clase con los nutrientes disponibles y la demanda ideal de la planta.
  49        
  50        :param nutrientes: Diccionario con los nutrientes y sus valores actuales en el suelo.
  51        :param demanda_planta: Valor total de la demanda nutricional ideal de la planta.
  52        """
  53        self.nutrientes = nutrientes
  54        self.demanda_planta = Decimal(demanda_planta)
  55
  56    def calcular_p(self, valor_registro: Decimal) -> Decimal:
  57        """
  58        Calcula el porcentaje de suficiencia de un nutriente con respecto a la demanda de la planta.
  59        
  60        :param valor_registro: Valor actual del nutriente en el suelo.
  61        :return: Porcentaje de suficiencia del nutriente con respecto a la demanda ideal.
  62        """
  63        if self.demanda_planta == 0:
  64            return Decimal('0.00')
  65        return (Decimal(valor_registro) / self.demanda_planta) * Decimal('100.00')
  66
  67    def calcular_i(self, mineral_p: Decimal, mineral_cv: Decimal) -> Decimal:
  68        """
  69        Calcula la cantidad de ajuste necesario para un nutriente limitante en función de su coeficiente de variación.
  70        
  71        :param mineral_p: Porcentaje de suficiencia del nutriente.
  72        :param mineral_cv: Coeficiente de variación del nutriente.
  73        :return: Cantidad de ajuste necesaria para alcanzar el nivel óptimo.
  74        """
  75        if mineral_p > Decimal('100.00'):
  76            result = ((mineral_p - Decimal('100.00')) * mineral_cv / Decimal('100.00'))
  77        else:
  78            result = ((Decimal('100.00') - mineral_p) * mineral_cv / Decimal('100.00'))
  79        return result.quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)
  80
  81    def calcular_r(self, mineral_p: Decimal, mineral_i: Decimal) -> Decimal:
  82        """
  83        Determina el nivel corregido del nutriente después del ajuste.
  84        
  85        :param mineral_p: Porcentaje de suficiencia del nutriente.
  86        :param mineral_i: Cantidad de ajuste aplicada al nutriente.
  87        :return: Nivel corregido del nutriente en el suelo.
  88        """
  89        if mineral_p > Decimal('100.00'):
  90            return (mineral_p - mineral_i).quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)
  91        return (mineral_p + mineral_i).quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)
  92
  93    def calcular_nutriente_limite(self, valores_registro: dict) -> str:
  94        """
  95        Identifica el nutriente más limitante según la Ley del Mínimo de Liebig.
  96        
  97        El nutriente limitante es aquel que tiene el menor porcentaje de suficiencia.
  98        
  99        :param valores_registro: Diccionario con los valores actuales de los nutrientes en el suelo.
 100        :return: Nombre del nutriente más limitante.
 101        """
 102        valores_p = {mineral: self.calcular_p(valor) for mineral, valor in valores_registro.items()}
 103        return min(valores_p, key=valores_p.get)  # Devuelve el nutriente con el menor porcentaje de suficiencia
 104
 105    def calcular_nutrientes(self, valores_registro: dict, valores_cv: dict) -> dict:
 106        """
 107        Calcula los ajustes necesarios para los nutrientes del cultivo.
 108        
 109        La corrección se aplica únicamente al nutriente más limitante para respetar la Ley del Mínimo de Liebig.
 110        
 111        :param valores_registro: Diccionario con los valores actuales de los nutrientes en el suelo.
 112        :param valores_cv: Diccionario con los coeficientes de variación de cada nutriente.
 113        :return: Diccionario con los valores de suficiencia (p), ajuste necesario (i) y nivel corregido (r) de cada nutriente.
 114        """
 115        nutriente_limitante = self.calcular_nutriente_limite(valores_registro)
 116        nutrientes = {}
 117        for mineral, valor_registro in valores_registro.items():
 118            p = self.calcular_p(valor_registro)
 119            i = self.calcular_i(p, valores_cv[mineral]) if mineral == nutriente_limitante else Decimal('0.00')
 120            r = self.calcular_r(p, i)
 121            nutrientes[mineral] = {'p': p, 'i': i, 'r': r}
 122        return nutrientes

 125class NutrientOptimizer:
 126    """
 127    Clase que optimiza la aplicación de productos para satisfacer los requerimientos de nutrientes de un cultivo,
 128    basada en la Ley del Mínimo de Liebig y programación lineal.
 129    """
 130
 131    def __init__(self, nutrientes_actuales: Dict[str, Decimal], demandas_ideales: Dict[str, Decimal], 
 132                 productos_contribuciones: Dict[str, Dict[str, Decimal]], coeficientes_variacion: Dict[str, Decimal]):
 133        """
 134        Inicializa el optimizador de nutrientes.
 135
 136        :param nutrientes_actuales: Diccionario con los niveles actuales de nutrientes (kg/ha o g/ha).
 137        :param demandas_ideales: Diccionario con los niveles ideales de nutrientes (kg/ha o g/ha).
 138        :param productos_contribuciones: Diccionario con los productos y sus contribuciones por nutriente.
 139        :param coeficientes_variacion: Diccionario con los coeficientes de variación por nutriente.
 140        """
 141        self.nutrientes_actuales = nutrientes_actuales
 142        self.demandas_ideales = demandas_ideales
 143        self.productos_contribuciones = productos_contribuciones
 144        self.coeficientes_variacion = coeficientes_variacion
 145        self.nutrientes = list(demandas_ideales.keys())
 146        self.productos = list(productos_contribuciones.keys())
 147
 148    def calcular_ajustes(self) -> Dict[str, Decimal]:
 149        """
 150        Calcula los ajustes necesarios para cada nutriente usando la Ley de Liebig adaptada.
 151        """
 152        ajustes = {}
 153        for nutriente in self.nutrientes:
 154            actual = self.nutrientes_actuales.get(nutriente, Decimal('0.0'))
 155            ideal = self.demandas_ideales[nutriente]
 156            if actual < ideal:
 157                p = (actual / ideal) * Decimal('100.0') if ideal > 0 else Decimal('0.0')
 158                i = ((Decimal('100.0') - p) * self.coeficientes_variacion[nutriente] / Decimal('100.0')).quantize(Decimal('0.00'), rounding=ROUND_HALF_UP)
 159                ajustes[nutriente] = (ideal - actual) * i  # Cantidad absoluta a ajustar
 160            else:
 161                ajustes[nutriente] = Decimal('0.0')
 162        return ajustes
 163
 164    def identificar_limitante(self) -> str:
 165        """
 166        Identifica el nutriente más limitante según la Ley de Liebig.
 167        """
 168        porcentajes = {
 169            nutriente: (self.nutrientes_actuales.get(nutriente, Decimal('0.0')) / self.demandas_ideales[nutriente]) * Decimal('100.0')
 170            if self.demandas_ideales[nutriente] > 0 else Decimal('0.0')
 171            for nutriente in self.nutrientes
 172        }
 173        return min(porcentajes, key=porcentajes.get)
 174
 175    def optimizar_productos(self) -> Tuple[Dict[str, Decimal], Dict[str, Decimal]]:
 176        """
 177        Optimiza las cantidades de productos a aplicar usando programación lineal.
 178        
 179        :return: Diccionario con las cantidades de productos y los nutrientes aportados.
 180        """
 181        ajustes = self.calcular_ajustes()
 182        
 183        # Coeficientes de la función objetivo (minimizar la suma de productos)
 184        c = [1] * len(self.productos)
 185
 186        # Matriz de restricciones (negativa para linprog)
 187        A_eq = []
 188        b_eq = []
 189        for nutriente in self.nutrientes:
 190            if ajustes[nutriente] > 0:
 191                fila = []
 192                for prod in self.productos:
 193                    contrib = self.productos_contribuciones[prod].get(nutriente, Decimal('0.0'))
 194                    fila.append(-float(contrib))  # Negativo para linprog
 195                A_eq.append(fila)
 196                b_eq.append(-float(ajustes[nutriente]))  # Negativo para linprog
 197
 198        # Límites (cantidades >= 0)
 199        bounds = [(0, None)] * len(self.productos)
 200
 201        # Resolver optimización
 202        res = linprog(c, A_eq=A_eq, b_eq=b_eq, bounds=bounds, method='highs')
 203        if not res.success:
 204            raise ValueError("No se pudo optimizar la aplicación de productos.")
 205
 206        # Resultados: cantidades de productos
 207        cantidades = {self.productos[i]: Decimal(str(round(x, 2))) for i, x in enumerate(res.x)}
 208
 209        # Calcular nutrientes aportados
 210        nutrientes_aportados = {nutriente: Decimal('0.0') for nutriente in self.nutrientes}
 211        for prod, cantidad in cantidades.items():
 212            for nutriente, contrib in self.productos_contribuciones[prod].items():
 213                nutrientes_aportados[nutriente] += contrib * cantidad
 214
 215        return cantidades, nutrientes_aportados

 217    def generar_recomendacion(self, lot_id: int) -> str:
 218        """
 219        Genera una recomendación para aplicar en el lote.
 220        
 221        :param lot_id: ID del lote donde se aplicará la recomendación.
 222        :return: Texto de la recomendación.
 223        """
 224        cantidades, nutrientes_aportados = self.optimizar_productos()
 225        lineas = [f"Aplicar en el lote {lot_id}:"]
 226        for prod, cantidad in cantidades.items():
 227            if cantidad > 0:
 228                lineas.append(f"- {cantidad} unidades de {prod}")
 229        lineas.append("Nutrientes aportados:")
 230        for nutriente, cantidad in nutrientes_aportados.items():
 231            unidad = "kg/ha" if nutriente in [n["name"] for n in macronutrients] else "g/ha"
 232            lineas.append(f"- {nutriente}: {cantidad} {unidad}")
 233        return "\n".join(lineas)

 297def calcular_cv_nutriente(lot_id, nutriente_name):
 298    """Determinar los Coeficientes de Variación """
 299    # Obtener valores históricos de LeafAnalysis para el lote
 300    valores = db.session.query(leaf_analysis_nutrients.c.value).join(LeafAnalysis).join(CommonAnalysis).filter(
 301        CommonAnalysis.lot_id == lot_id,
 302        leaf_analysis_nutrients.c.nutrient_id == Nutrient.query.filter_by(name=nutriente_name).first().id
 303    ).all()
 304    valores = [v[0] for v in valores]
 305    if len(valores) < 2:
 306        return Decimal('0.5')  # Valor por defecto si no hay suficientes datos
 307    mu = mean(valores)
 308    sigma = stdev(valores)
 309    return Decimal(str(sigma / mu)).quantize(Decimal('0.01'))

 320def determinar_coeficientes_variacion(lot_id: int) -> Dict[str, Decimal]:
 321    coeficientes = {}
 322    nutrientes = [n["name"] for n in macronutrients + micronutrients]
 323    for nutriente in nutrientes:
 324        cv = calcular_cv_nutriente(lot_id, nutriente)
 325        if cv == Decimal('0.5'):  # Valor por defecto si no hay datos
 326            # Asignar valores basados en literatura
 327            if nutriente in ["Nitrógeno"]:
 328                cv = Decimal("0.5")
 329            elif nutriente in ["Fósforo"]:
 330                cv = Decimal("0.3")
 331            elif nutriente in ["Potasio"]:
 332                cv = Decimal("0.4")
 333            elif nutriente in ["Cobre", "Zinc"]:
 334                cv = Decimal("0.25")
 335            else:
 336                cv = Decimal("0.3")  # Default genérico
 337        coeficientes[nutriente] = cv
 338    return coeficientes

 404    def _process_objectives_by_crop(self, objectives):

 406        crop_dict = {}
 407        for obj in objectives:
 408            serialized = self._serialize_objective(obj)
 409            crop_name = serialized["crop_name"].lower()  # e.g., 'arroz', 'papa'

 416            nutrient_dict = {

 419            }
 420            # Add objective data to the crop's list

 422                "id": serialized["id"],
 423                "created_at": serialized["created_at"],
 424                "updated_at": serialized["updated_at"],
 425                "nutrients": nutrient_dict

 442class CropObjectives:

 447    def get(self, index=None, id=None):
 448        """Access a specific objective by index or id"""
 449        if id is not None:
 450            for obj in self.objectives:
 451                if obj["id"] == id:
 452                    return CropData(obj["nutrients"])
 453            raise ValueError(f"No objective found with id {id}")
 454        if index is not None:

 456                return CropData(self.objectives[index]["nutrients"])

 458        # Default: return the most recent objective (based on updated_at)
 459        sorted_objectives = sorted(self.objectives, key=lambda x: x["updated_at"], reverse=True)
 460        return CropData(sorted_objectives[0]["nutrients"])

 60

179

181        """
182        ajustes = self.calcular_ajustes()

184        # Coeficientes de la función objetivo (minimizar la suma de productos)

188        A_eq = []

def findLimitingNutrient():
    limitingNutrient = None
    lowestPercentage = 100

    for nutrient, value in analysisData["foliar"].items():
        if nutrient in optimalLevels["foliar"]:
            min_value = optimalLevels["foliar"][nutrient]["min"]
            max_value = optimalLevels["foliar"][nutrient]["max"]
            optimalMid = (min_value + max_value) / 2
            percentage = (value / optimalMid) * 100
            if percentage < lowestPercentage and percentage < 90:
                lowestPercentage = percentage
                limitingNutrient = {
                    "name": nutrient,
                    "value": value,
                    "optimal": optimalMid,
                    "percentage": percentage,
                    "type": "foliar",
                }

    for nutrient, value in analysisData["soil"].items():
        if nutrient in optimalLevels["soil"] and nutrient != "ph":
            min_value = optimalLevels["soil"][nutrient]["min"]
            max_value = optimalLevels["soil"][nutrient]["max"]
            optimalMid = (min_value + max_value) / 2
            percentage = (value / optimalMid) * 100
            if percentage < lowestPercentage and percentage < 90:
                lowestPercentage = percentage
                limitingNutrient = {
                    "name": nutrient,
                    "value": value,
                    "optimal": optimalMid,
                    "percentage": percentage,
                    "type": "soil",
                }

    return limitingNutrient
