 111def check_resource_access(resource, claims):
 112    """
 113    Verifica si un usuario tiene acceso a un recurso espec√≠fico basado en su rol y claims.
 114    
 115    Args:
 116        resource: El recurso al que se intenta acceder (debe tener un atributo org_id)
 117        claims: Diccionario con informaci√≥n del usuario (rol, user_id, org_id)
 118    
 119    Returns:
 120        bool: True si el usuario tiene acceso, False en caso contrario
 121    
 122    Raises:
 123        ValueError: Si los claims no contienen la informaci√≥n necesaria
 124    """
 125    user_role = claims.get("rol")
 126    if not user_role:
 127        return False
 128
 129    # Caso del rol ADMINISTRATOR: acceso total
 130    if user_role == RoleEnum.ADMINISTRATOR.value:
 131        return True
 132    
 133    # Caso del rol RESELLER: acceso a recursos de organizaciones en su paquete
 134    if user_role == RoleEnum.RESELLER.value:
 135        reseller_org_id = claims.get("org_id")
 136        if not reseller_org_id:
 137            return False
 138        reseller_package = ResellerPackage.query.filter_by(reseller_id=reseller_org_id).first()
 139        if not reseller_package:
 140            return False
 141        resource_org_id = getattr(resource, "org_id", None)
 142        if resource_org_id is None:
 143            return False
 144        return any(org.id == resource_org_id for org in reseller_package.organizations)
 145    
 146    # Caso de roles organizacionales: ORG_ADMIN, ORG_EDITOR, ORG_VIEWER
 147    if user_role in (RoleEnum.ORG_ADMIN.value, RoleEnum.ORG_EDITOR.value, RoleEnum.ORG_VIEWER.value):
 148        user_id = claims.get("user_id")
 149        if not user_id:
 150            return False
 151        user = User.query.get(user_id)
 152        if not user:
 153            return False
 154        resource_org_id = getattr(resource, "org_id", None)
 155        if resource_org_id is None:
 156            return False
 157        return any(org.id == resource_org_id for org in user.organizations)
 158    
 159    return False

 165    def post(self):
 166        """User login endpoint
 167        :param str username: Registered username
 168        :param str password: Account password
 169        :status 200: Successful login
 170        :status 400: Invalid request data
 171        :status 401: Authentication failure
 172        """
 173        try:
 174            data = request.get_json()
 175            if not data:
 176                return jsonify({"msg": "Missing request data"}), 400
 177
 178            username = data.get("username", "").strip()
 179            password = data.get("password", "").strip()
 180
 181            if not username or not password:
 182                return jsonify({"msg": "Credentials required"}), 400
 183
 184            user = User.get_by_username(username)
 185            if self._invalid_credentials(user, password):
 186                return jsonify({"msg": "Invalid credentials"}), 401
 187
 188            claims = self._build_claims(user)
 189            tokens = self._generate_tokens(str(user.id), claims)
 190            response = self._build_response(tokens, claims)
 191
 192            return response
 193
 194        except Exception as e:
 195            print(f"Login error: {str(e)}")
 196            return jsonify({"msg": "Invalid request"}), 400
 197
 198    def _invalid_credentials(self, user, password):
 199        """Validate user credentials"""
 200        return not user or not user.check_password(password) or not user.active
 201
 202    def _build_claims(self, user):
 203        """Construct JWT claims payload"""
 204        claims = {
 205            "id": user.id,
 206            "username": user.username,
 207            "email": user.email,
 208            "organizations": [
 209                {"id": org.id, "name": org.name} for org in user.organizations
 210            ],
 211            "rol": user.role.value,
 212        }
 213
 214        if user.role == RoleEnum.RESELLER:
 215            reseller_package = user.reseller_packages.first()
 216            if reseller_package:
 217                claims["reseller_organizations"] = [
 218                    {"id": org.id, "name": org.name}
 219                    for org in reseller_package.organizations
 220                ]
 221        return claims
 222
 223    def _generate_tokens(self, identity, claims):
 224        """Generate JWT tokens"""
 225        return (
 226            create_access_token(identity=identity, additional_claims=claims),
 227            create_refresh_token(identity=identity, additional_claims=claims),
 228        )
 229
 230    def _build_response(self, tokens, claims):
 231        """Build final response with cookies"""
 232        access_token, refresh_token = tokens
 233        response = jsonify(
 234            {
 235                "access_csrf": get_csrf_token(access_token),
 236                "refresh_csrf": get_csrf_token(refresh_token),
 237                "additional_claims": claims,
 238                "msg": "Authentication successful",
 239            }
 240        )
 241        set_access_cookies(response, access_token)
 242        set_refresh_cookies(response, refresh_token)
 243        return response
